Internal function to check summary stats

Usage

check.summ.stats(stats)

Arguments

stats
a named list of summary statistics

Internal function to check summary stats

Value

named list of summary statistics after being checked.

Description

Makes sure summary statistics sent to summ.stats are in the correct format

Examples

## produce list of summary stats (using default summary statistics)
check.summ.stats(def.summ.stats())

$s.var
function (unit.tree) 
{
    assert.is.unit.tree(unit.tree)
    sd.pic <- sqrt(unit.tree$pics[, "variance"])
    abs.pic <- abs(unit.tree$pics[, "contrasts"])
    cv <- lm(abs.pic ~ sd.pic)
    unname(coef(cv)["sd.pic"])
}

$s.hgt
function (unit.tree) 
{
    assert.is.unit.tree(unit.tree)
    ht <- edge.height(unit.tree$phy)
    N <- Ntip(unit.tree$phy)
    nh <- ht$end[c((N + 1):nrow(ht))]
    names(nh) <- rownames(ht[c((N + 1):nrow(ht)), ])
    abs.pic <- abs(unit.tree$pics[, "contrasts"])
    ch <- lm(abs.pic ~ nh)
    unname(coef(ch)["nh"])
}
